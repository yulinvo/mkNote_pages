# String.prototype.matchAll

matchAll方法用来得到正则批量匹配的结果

let str = `<ul>
        <li>
            <a>肖生克的救赎</a>
            <p>上映日期: 1994-09-10</p>
        </li>
        <li>
            <a>阿甘正传</a>
            <p>上映日期: 1994-07-06</p>
        </li>
    </ul>`;

//声明正则
const reg = /<li>.*?<a>(.*?)<\/a>.*?<p>(.*?)<\/p>/sg

//调用方法
const result = str.matchAll(reg);

// for(let v of result){
//     console.log(v);
// }

const arr = [...result];
console.log(arr);

# 类的私有属性

使用#age这种方式定义类的私有属性，私有属性只能在类的内部使用，不能在外部调用

class Person {
    //公有属性
    name;
    //私有属性
    #age;
    #weight;
    //构造方法
    constructor(name, age, weight) {
        this.name = name;
        this.#age = age;
        this.#weight = weight;
    }

    intro() {
        console.log(this.name);
        console.log(this.#age);
        console.log(this.#weight);
    }
}

//实例化
const girl = new Person('晓红', 18, '45kg');

console.log(girl.name);
// console.log(girl.#age);
// console.log(girl.#weight);

girl.intro();

# Promise.allSettled

Promise.allSettled接收一个promise的数组，返回的也是一个promise的对象，不过返回的结果永远是一个成功的状态，成功的值是每一个promise的状态和结果

跟Promise.allSettled很像的是Promise.all方法，区别是allSettled永远成功，all方法是全部成功才成功，只有有任何一个失败则all的状态就为失败

另外all的结果：都成功了是所有成功值的【数组】，有失败的话则返回失败的结果（非数组，忽略掉成功的值）

//声明两个promise对象
const p1 = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('商品数据 - 1');
    }, 1000)
});

const p2 = new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('商品数据 - 2');
        // reject('出错啦!');
    }, 1000)
});

//调用 allsettled 方法
// const res = Promise.allSettled([p1, p2]);     //res状态始终成功

const res = Promise.all([p1, p2]);       //当有失败的时候res的结果就不是数组了，而是十失败的

console.log(res);

# 动态 import 导入

静态导入：import * as m1 ...

动态导入：使用import()函数，import()函数的返回结果是一个promise对象，结果中的数据类似于通用的导入方式import * as m1 ...中的m1

index.html

<body>   
    <button id="btn">点击</button>
    <script src="./js/app.js" type="module"></script>
</body>

app.js        //动态引入

// import * as m1 from "./hello.js";            //静态引入

const btn = document.getElementById('btn');
btn.onclick = function(){
    import('./hello.js').then(module => {        //动态引入
        module.hello();
    });
}

hello.js

export function hello(){
    alert('Hello');
}

# BigInt

表示方法：在普通数字的基础上加一个n，如100n

bigint类型的不能和普通类型的数字直接运算，如bigint(100) + 1会报错

//直接定义大整形
// let n = 521n;

//大数值运算
let max = Number.MAX_SAFE_INTEGER;
console.log(max);
console.log(max + 1);
console.log(max + 2);                //这样加下去会有问题
