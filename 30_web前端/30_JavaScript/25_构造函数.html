# 创建构造函数

工厂函数写法：

function createPerson(name , age , gender){
	var obj = {
		name: name,
		age: age,
		gender: gender,
		sayName: function(){
			alert(this.name);
		}
	}
	return obj
}

var p1 = createPerson('tom',12,'male')			//直接调用，不需要使用new关键字
var p2 = createPerson('jim',15,'male')
//缺点：p1、p2是Object类型（对象没有一个具体的类型），无法区分createPerson、createDog之类的

构造函数通常的写法：

function Person(name , age , gender){
    this.name = name;
    this.age = age;
    this.gender = gender;
    this.sayName = function(){			#缺点：每个对象都有相同的方法，浪费内存
        alert(this.name);
    };
}

var per = new Person(‘tom’,20,'male')

console.log(per instanceof Person);		#检查一个对象是否是一个类的实例，如果是，则返回true，否则返回false：
console.log(per instanceof Object);		#object也是顶级的类，所以任何对象和Object做instanceof检查时都会返回true

# 将构造函数中fun提取出来

如果将方法写在构造函数中，每次实例化都会开辟一块内存区域，对资源是一种浪费

function Person(name , age , gender){
    this.name = name;
    this.age = age;
    this.sayName = function(){...};
}
let p1 = new Person('tom',20,'male')
let p2 = new Person('jerry',20,'male')

弊端： 
	构造函数每次实例化，this.name、this.age开辟空间，因为只是一个变量倒还好；
	this.sayName是一个函数，每次都会单独开辟一个内存区域，检查：console.log(p1.sayName == p2.sayName)结果为false

  1. 改进步骤一：将sayName方法提取到全局，这样就只会创建一次sayName方法

    function fun2(){...};
function Person(name,age,gender){
    this.name = name;
    this.age = age;
    this.sayName = fun2;
}
检查：console.log(p1.sayName == p2.sayName)结果为true

    弊端：将函数定义在全局作用域中，污染了全局作用域的命名空间，而且定义在全局作用域中也不安全（多人开发时不留意也定义了一个fun2的函数时则之前的功能就被覆盖了）

  2. 改进步骤二： 将sayName从提取到全局，改到变换至原型上：

    Person.prototype.sayName = function(){...}

检查：console.log(p1.sayName == p2.sayName)结果为true

# 原型对象

  • 普通函数的原型对象

    每一个函数（包括普通函数），解析器都会向函数中添加一个属性prototype，如果函数作为普通函数调用prototype没有任何作用

    function fn1(){
	console.log('fn1....');
}
console.log('fn1:',fn1.prototype);				//普通函数也有prototype属性

    类的显示原型对象prototype、实例的隐式原型对象__proto__

# hasOwnProperty

使用in检查对象中是否含有某个属性时，如果对象中没有但是原型中有，也会返回true

可以使用对象的hasOwnProperty()来检查对象自身中是否含有该属性，使用该方法只有当对象自身中含有属性时，才会返回true

function Person(){ }
Person.prototype.name = "我是原型对象中的name"

let p1 = new Person()
console.log("name" in p1);		#使用in检查对象中是否含有某个属性时，如果对象中没有但是原型中有，也会返回true ！！！！
console.log(p1.hasOwnProperty("name"));			#只检查自身，不检查原型！

# 静态(类)属性、静态(类)方法

  • 静态属性类似于class中的static name='...'
  • 静态方法类似于class中的static change(){}
function Phone(){
    
}
Phone.name = '手机'
Phone.change = function(){
    console.log("我可以改变世界");
}

let nokia = new Phone();
console.log(nokia.name)					//实例中读取不到静态属性
console.log(nokia.change)				//实例中读取不到静态方法

对应的class中的写法为：

class Phone {
    static name = "手机"
    static change() {
        console.log("我可以改变世界");
    }
}
let nokia = new Phone()
console.log(nokia.name)							//实例读取不到静态属性
console.log(nokia.change)						//实例读取不到静态方法

注意：如果static name="手机"如果不写static实例能读取到name属性，实例中属性的来源：1：constructor中定义this.xx=... 2 类中定义class Phone{ name="手机" }的

# 继承

//手机
function Phone(brand, price){
    this.brand = brand;
    this.price = price;
}

Phone.prototype.call = function(){
    console.log("我可以打电话");
}

//智能手机
function SmartPhone(brand, price, color, size){
    Phone.call(this, brand, price);				//通过call方法改变调用对象；相比与继承，这一步其实是在实例化！！！！
    this.color = color;
    this.size = size;
}

//设置子级构造函数的原型
SmartPhone.prototype = new Phone;					//!!!!!!!!!!!!  SmartPhone.prototype为{constructor: ƒ}
	//此处使用 new Phone 或者是 new Phone()都是一样的，结果都是{}，测试：p1=new Phone && p2=new Phone()，打印比较下
SmartPhone.prototype.constructor = SmartPhone;

//声明子类的方法
SmartPhone.prototype.photo = function(){
    console.log("我可以拍照")
}

SmartPhone.prototype.playGame = function(){
    console.log("我可以玩游戏");
}

const chuizi = new SmartPhone('锤子',2499,'黑色','5.5inch');

console.log(chuizi);

原型对象protype 指向的是一个类似0x123的地址： SmartPhone.prototype = new Phone;
