# 搭建json-server

github地址：https://github.com/typicode/json-server

搭建步骤：

  1. Install JSON Server

    npm install -g json-server

  2. Create a db.json file with some data

    {
  "posts": [
    { "id": 1, "title": "json-server", "author": "typicode" },
    { "id": 2,"title": "上硅谷大厂学院上线了","author": "小编" },
	{ "id": 3,"title": "今天天气不错，还挺风和日丽的","author": "张三" }
  ],
  "comments": [
    { "id": 1, "body": "some comment", "postId": 1 }
  ],
  "profile": { "name": "typicode" }
}

  3. Start JSON Server

    json-server --watch db.json

Now if you go to http://localhost:3000/posts/1, you'll get

{ "id": 1, "title": "json-server", "author": "typicode" }

# axios的默认配置

全局的 axios 默认值

axios.defaults.method = 'GET';
axios.defaults.params = {id:100};		#请求变为了：  xxx/posts?id=100
axios.defaults.timeout = 3000;

axios.defaults.baseURL = 'http://localhost:3000';  		// baseURL将自动加在url前面，除非url是一个绝对 URL。
axios.defaults.headers.common['Authorization'] = AUTH_TOKEN;
axios.defaults.headers.post['Content-Type'] = 'application/x-www-form-urlencoded';

设置了默认配置后，axios可以简化如下：

axios({
	method: "GET",
	url: 'http://localhost:3000/posts'
}).then(r=>console.log(r))

	url: '/posts'
}).then(r=>console.log(r))

# 拦截器

拦截器实际就是promise中的then方法，接收成功、失败的两个回调

# 初识拦截器

// 添加请求拦截器
axios.interceptors.request.use(function (config) {
    console.log('请求拦截器 -- 成功')
    throw '参数出了问题'
    // return config;
}, function (error) {
    console.log('请求拦截器 -- 失败')
    return Promise.reject(error);
});

// 添加响应拦截器
axios.interceptors.response.use(function (response) {
    console.log('响应拦截器 -- 成功')
    return response;
}, function (error) {
    console.log('响应拦截器 -- 失败')
    return Promise.reject(error);
});

axios.get('/posts').then(v => {
    console.log('自定义函数处理--成功')
}, r => {
    console.log('自定义函数处理--失败')
}).catch( r => {
    console.log('自定义函数处理--失败 catch')       //then中有失败处理时catch就捕获不到失败了
})

结果：
请求拦截器 -- 成功
响应拦截器 -- 失败
自定义函数处理--失败

    console.log('响应拦截器 -- 成功 2号')
    return response;
}, function (error) {
    console.log('响应拦截器 -- 失败 2号')

});

<head>
    <link crossorigin="anonymous" href="https://cdn.bootcdn.net/ajax/libs/twitter-bootstrap/3.3.7/css/bootstrap.min.css" rel="stylesheet">
    <script src="https://cdn.bootcdn.net/ajax/libs/axios/0.21.1/axios.min.js"></script>
</head>

<body>
    <div class="container">
        <h2 class="page-header">axios取消请求</h2>
        <button class="btn btn-primary">发送请求</button>
        <button class="btn btn-warning">取消请求</button>
    </div>
    <script>
        axios.defaults.timeout = 5000
        const btns = document.querySelectorAll('button')
        let cancel = null

        btns[0].onclick = function () {
            if(cancel !== null){
                cancel()
            }
            axios.get('http://localhost:3000/posts', {
                cancelToken: new axios.CancelToken(function executor(c) {
                    // executor 函数接收一个 cancel 函数作为参数
                    cancel = c;
                })
            }).then(v => {
                cancel = null;
            })
        }

         btns[1].onclick = function () {
            cancel()
         }
    </script>
</body>
