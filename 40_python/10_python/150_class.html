# 类方法

类方法同类属性一样也是跟类本身相关，跟具体的对象无关，使用cls.xx访问类属性、访问其他类方法！！！！！！！！

参考文档：https://www.bilibili.com/video/BV1ex411x7Em?p=412&spm_id_from=pageDriver

类方法只能访问类变量，不能访问实例变量（如self.name,self.age）

class A:
    count = 100
    
    @classmethod
    def demo1(cls):
        print('demo1----')
        
    @classmethod
    def demo2(cls):
        print('demo2----')
        print(cls.count)					#通过cls调用类属性
        cls.demo1()							#通过cls调用类方法
        print(cls,cls())				    #<class '__main__.A'> <__main__.A object at 0x000001CB9FA27F10>
        	#print(cls)和print(A)的结果是一样的
            #cls()和a1=A()然后print(a1)结果是类似的，cls()应该就是实例化吧？
        return cls()						#会把这个class类传递过去

A.demo2()

a1 = A()
a1.demo2()					#实例化后也能正常调用函数，对应了django中的View--as_view()--classonlymethod

将一个没有关键字的方法（self不算）变成一个属性

property属性方法传参 实例调用属性方法不加（），所以调用时传参有点特殊

class Dog(object):
    n=100
    def __init__(self,name):
        self.name = name
        self.__food = None              #用于存储传递的food参数

    @property           #把一个方法变成静态属性，如果需要传递参数呢，如 def eat(self,food):
    def eat(self):
        print("%s is eating %s"%(self.name,self.__food))

    @eat.setter
    def eat(self,food):                 #写的方法名还必须是eat
        print('set to food:',food)
        self.__food = food

    @eat.deleter
    def eat(self):
        del self.__food
        print('删完了')

d = Dog('xx')
d.eat
d.eat = 'baozi'         #给属性赋值'baozi'
d.eat

del d.eat

#d.eat					#在查看已经没有该属性了

# 私有属性、私有方法

以 __ 开始即可

私有属性、方法只能在class类内部进行调用，无法在类的外部调用（继承后也无法调用）

参考文档：https://www.bilibili.com/video/BV1ex411x7Em?p=389&spm_id_from=pageDriver

class Women:
    __a = 111									#类 私有变量
    def __init__(self,name) -> None:
        self.name = name
        self.__age = 18
    
    def __secret(self):
        print("%s 的年龄是 %d,类变量是 %d" %(self.name,self.__age,self.__a))     #内部可以直接调用私有变量

xiaofang = Women("小芳")
print(xiaofang._Women__age)						#实例 强制调用私有变量
xiaofang._Women__secret()						#实例 强制调用私有方法


print(xiaofang._Women__a)						#实例 强制调用类变量
print(Women._Women__a)							#类   强制调用类变量

# 单实例设计模式代码

正确的代码，多次创建实例返回的都是相同的内存地址：
    class A:
        instance = None
        init_flag = False

        def __init__(self) -> None:				#init动作只执行一次，不是必须的，可以不写
            if A.init_flag:
                return
            self.name = 10  					#正常的初始化动作...
            A.init_flag = True

        def __new__(cls,*args,**kwargs):			#只分配一次内存空间，这个是必须的
            if cls.instance is  None:
                cls.instance = super().__new__(cls, *args, **kwargs)   #new是一个静态方法，所以需要传递一个参数cls
            return cls.instance						#注意这个编码逻辑，多用这种方式，不要写一堆if/else

    a1 = A()
    a2 = A()            #应该是会覆盖a1的初始化，没有详细测试

    print(a1, a2)

# _add_、__mul__、__abs__

class MyInt(object):
    def __init__(self, num):
        self.num = num
    def __add__(self, n):
        return self.num+n
    def __mul__(self, n):
        return self.num*n
    def __str__(self):
        return "a的值是: %d" % (self.num)
    def __abs__(self):
        print('in abs')

a = MyInt(10)

a+10                #调用__add__
a*10                #调用__mul__
print a             #调用__str__
abs(a)              #abs是python内置的方法; n=10,dir(n)就有__abs__方法，n.__abs__()效果同abs(n)
