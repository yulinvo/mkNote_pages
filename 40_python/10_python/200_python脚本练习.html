#coding:utf-8
import threading,time
def run(n):
    print 'task',n
    time.sleep(2)
    print 'task_done',n
start_time = time.time()
t_obj= []
for i in  range(50):
    t = threading.Thread(target=run,args=('t-%s' %i,))
  t.setDaemon(True) #将当前线程设置为守护线程，一定要在start之前设置    守护线程的意思是当主线程结束后，子线程也强制结束了，效果是加上这句后fun函数里的print 'task_done',不会执行了
    t.start()                   #子线程执行完run函数后就退出了，不会一直挂着
#t.join()               #此时t是最后的i为49的时候的t，为什么t能从for循环传递出来呢？？？  原因：函数执行完变量就释放了，但此处是循环不是函数，执行完后变量不会释放
    t_obj.append(t)
print 't_onj:' ,t_obj
for t in t_obj:
    t.join()             #子线程没有执行完之前，主线程会卡住；在此处的意思是50个子线程全部执行完后，才执行主线程后面的代码
print 'cost:',time.time() - start_time ,threading.current_thread(),threading.active_count()         #这里不会得到所有子进程执行完函数run后一共用了多少时间，得到的是启动完所有子线程用的时间，线程和子线程之间是独立的，threading.current_thread()为当前进程所在位置（主线程会提示_MainThread(MainThread），threading.active_count()是统计当前活跃的进程个数

#coding:utf-8
import  threading,time
event = threading.Event()
def lighter():
    count = 0
    event.set()         #先设置绿灯
    while True:
        if count > 5 and count < 10:        #改为红灯
            event.clear()                    #把标志位清了
            print '\033[41;1m red light is on...\033[0m'
        elif count > 10:
            event.set()                      #变绿灯
            count = 0
        else:
            print '\033[42;1m green light is on...\033[0m'
        time.sleep(1)
        count += 1
def car(name):
    while True:
        if event.is_set():          #代表绿灯，

        else:
            print '[%s] sees red light ,waiting...'% name
            event.wait()
            print '\033[34;1m [%s] green light is on,start going... \033[0m'% name

#coding:utf-8
import SocketServer,socket
class MyTCPHandle(SocketServer.BaseRequestHandler):
    def handle(self):                       #跟客户端所有的交互过程都是handle函数处理的
        while True:
            try:
                self.data = self.request.recv(1024).strip()       # self.request：客户端每一个请求过来都会MyTCPHandle实例化; 客户端不传递数据 recv就在这等着，等待时默认是阻塞的
                print '{} wrote:'.format(self.client_address[0])            # self.client_address[0] 客户端的ip地址
                print self.data
                if not self.data:                   #客户端不是不传了，不传了会阻塞在self.request.recv(1024).strip，没有阻塞又收到的数据是空 表示断开了
                    print '客户端已断开'
                    break                           # break这个实例就被释放了
                self.request.sendall(str(self.data.upper()))   # sendall重复调用send，此处也可以使用send
            except Exception as e:
                print 'err:',e
                break
if __name__ == '__main__':
    HOST,PORT = 'localhost',9999
    server = SocketServer.TCPServer((HOST,PORT),MyTCPHandle)
    # server = SocketServer.ThreadingTCPServer((HOST,PORT),MyTCPHandle)                #多线程
    # server = SocketServer.ForkingTCPServer((HOST,PORT),MyTCPHandle)             #多进程（winodws上有点问题，linux上没问题）
    server.serve_forever()                         # 这个可以同时处理多个请求，多线程？？？

# yield练习

def cash_money(amount):
    while amount > 0:
        amount -= 100
        yield 100               #执行到yield程序就中断执行了，100是执行atm.next()时返回100,接收值通过别人的send方法
        print '又来取钱了！'

print type(atm)          #输出结果：<type 'generator'> ,而type(cash_money)只是一个普通函数<type 'function'>

# 寻找差异

数据库中原有： old_dict={ '#1':{'hostname':'c1','cpu_count':2,'mem_capicity':80}, '#2':{'hostname':'c1','cpu_count':2,'mem_capicity':80}, '#3':{'hostname':'c1','cpu_count':2,'mem_capicity':80} } new_dict={ '#1':{'hostname':'c1','cpu_count':2,'mem_capicity':800}, '#3':{'hostname':'c1','cpu_count':2,'mem_capicity':80}, '#4':{'hostname':'c1','cpu_count':2,'mem_capicity':80} }

代码如下：

修改old_dict中的值，使其和最新的保持一致：
old_dict={
    '#1':{'hostname':'c1','cpu_count':2,'mem_capicity':80},
    '#2':{'hostname':'c1','cpu_count':2,'mem_capicity':80},
    '#3':{'hostname':'c1','cpu_count':2,'mem_capicity':80}
}
new_dict={
    '#1':{'hostname':'c1','cpu_count':2,'mem_capicity':800},
    '#3':{'hostname':'c1','cpu_count':2,'mem_capicity':80},
    '#4':{'hostname':'c1','cpu_count':2,'mem_capicity':80}
}
old_keys= set(old_dict.keys())
new_keys=set(new_dict.keys())

update_set=old_keys.intersection(new_keys)
delete_set=old_keys.symmetric_difference(update_set)
add_set=new_keys.symmetric_difference(update_set)
for i in update_set:
    old_dict[i]=new_dict[i]
for i in delete_set:
    old_dict.pop(i)
for i in add_set:
    old_dict[i]=new_dict[i]
print old_dict

    username=raw_input('pls input your account>>>').strip()

    print retry_dict
