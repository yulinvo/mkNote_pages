# Error和Exception

异常范围的树状结构图：
    Error的一般是错误，系统已经明确知道是错误而不是异常；
    exception的一般是异常，系统不知道是什么问题，没有出现预期的结果

names = ['a','b']
data = {}

try:                        #try里面的语句只要有报错就不会往下执行了
    names[2]
    data['name']

except IndexError as e:             #except会根据报错信息逐条查找，直到找到匹配的错误类型；另外except IndexError：print'xxx'也行，只是这样就写死了，
    print 'IndexError:',e
except KeyError as e:               # except 里的语句常有一个break
    print 'KeyError',e
except  Exception as e:
    print "未知错误",e
else:                               #当没有任何错误的时候(try里面的都正常)，这行这个
    print '一切正常'
finally:
    print '不管有没有错误，都执行！'
print 'xxxx'                        # 后面的语句依然能够正常执行

try:
    num = int(input("请输入一个整数: "))
    result = 8/num
    print(result)
except ValueError:
    print("请输入一个整数，而不是字母")
except ZeroDivisionError as e:					#具体的异常上也能使用as e这种写法
	print(e,type(e),dir(e))						#查看异常中都有哪些方法
    print('数字0不能作为分母')
except Exception as e:							#捕获未知异常
    print(e)
else:
    print("没有出现异常，程序正常执行；'---没有异常才会执行的代码---'")
finally:
    print("无论是否出现异常，都会执行的代码")

----------------------------------------------------------------------------
except (ValueError, ZeroDivisionError):					#同时捕获两个异常
    print("输入数字类型不合法")

except:  		#和except Exception as e相比，这种写法不能捕获具体的异常信息，只能输出用户自定义的异常信息，就是下面这句
    print("请输入一个整数，而不是字母")

# 自定义异常

参考文档：https://www.bilibili.com/video/BV1ex411x7Em?p=431&spm_id_from=pageDriver

def input_password():
    pwd = input("请输入密码:")

    if len(pwd) >=8 :
        return pwd
    print("主动抛出异常")
    ex = Exception("密码长度不够") 						#定义一个异常实例
    # ex = Exception("密码长度不够","你好")  			   #Exception接收的是一个*args参数
    raise ex									#抛出异常实例，异常的参数是固定的，下面的例子中异常的参数是传递的

try:
    input_password()
except Exception as e:
    print(e)
