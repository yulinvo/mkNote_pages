# python内置函数

abs():取绝对值

all():传递参数中每一个都是真，结果才为True ;
	0、none、空字符串、空列表、空元组、空字典为假，all([11,22,33,''])为假，去除‘’后为真(‘ ’只有一个空格也是真的)；
	all([])真，all([''])假，判断里面的元素？  可以使用bool查看真假：bool(None)、bool(‘’)
any():只要有一个为真的，返回真      
	any(['',None,[],{},1]):没有1则为假，红色部分似乎解释了all中all([])真(这里面的[]是固定用法)，all([''])假的问题

assert :断言，也可以使用if判断    assert  type('aaa') is str ; print 123

map():映射，map的第一个参数是函数，func，没有()    如：li=[11,22,33];print map(lambda x:x*2,li) ,结果[22, 44, 66]

# 函数传参

  • 指定参数名称

    def main(a1,a2):
	print a1,a2

main(a2=1,a1=2)				#可以颠倒参数位置

  • 默认参数（默认参数必须要放在最后面，否则会报语法错误）

    def main(a1,a2,a3='abc'):	#c如果不传递使用'abc'，传递则以传递的参数为准；如果此处b有默认值则c必须要有默认值；
	print a1,a2,a3

main(1,2)

    def main(a=10,b=20,c=30):
    print(a,b,c)
    
main(1,2)						#1,2,30
main(a=2,b=1)					#2,1,30

​

  • 动态参数（*放到前面，**放到后面，顺序颠倒会报错，语法规定这样）

    def main(*args,**kwargs):
    print args,kwargs  	  #args为tuple类型:(11, 22, 33)   kwargs为dict类型：{'uu': 22, 'n1': 11, 'xx': 33} 
    
    for i in args:
        print i

main(11,22,33,n1=11,uu=22,xx=33)

    • 当传递预先定义的list、dict时

      def main(*args,**kwargs):
	pass

a1=[11,22,33]
a2={'n1':11,'uu':22,'xx':33}

main(a1,a2)		#会将a1和a2作为一个tuple都传递给args了，kwargs里面是空的；正确的方法是：main(*a1,**a2)

# 递归函数

一：
    def func(a):
        print a
        if a/2 >1:
            res=func(a/2)
            print res      #满足(a/2>1)条件时不会执行这句；执行多少层，最后就退出多少层，func到res=func(a/2)之间没有
                             return，所以返回None
            return res      #满足(a/2>1)条件时不会执行这句；直到最后跳出循环才执行
    func(100)



二：
    def func(a):
        print a
        if a/2 >1:
            return func(a/2)        #这种加上一个return是自结束，就不会有那么多层的None
    func(100)



三:
    def calc(a):
        print a
        if a/2 > 1:
            res = calc(a/2)
            print ('res',res)
        print "A",a
        return  a          #return是返回给上级函数，返回给了res = calc(a/2) 这里
    calc(10)
