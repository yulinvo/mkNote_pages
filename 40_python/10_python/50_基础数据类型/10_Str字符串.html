Str字符串

# 字符串类型判断

name.isalnum()     			#是否是字母或者数字
name.isalpha()     			#是否是字母
name.isdecimal()     		#是否是十进制小数     2.7好像没有
name.isdigit()     			#是否是数字
name.isidentifier()     	#好像是是否是关键字之类的
name.islower()     			#是否全部小写
name.isnumeric()     		#是否是数字
name.isprinttable()     	#是否可以打印tab么？？
name.isspace()          	#是否是空格
name.istitle()          	#给定的字符串是否是标题（标题通常首字母都是大写）
name.isupper()          	#是否全部大写

# len

长度：len(obj)

# 索引

索引：obj[1]     #str[1]

# 切片

切片：obj[1:]、obj[1:10]     #str[1:5]

# 大小写

首字母：
    name.capitalize()           #首字母大写
    name.casefold()             #首字母小写
    name.swapcase()             #首字母大写小写互换

操作整个字符串：
	name.upper()                #全部转换为大写
	name.lower()                #全部转换为小写

# encode

name.encode('gbk')               #编码

# count

name.count('f',3,10)               #字符'f'字符出现的次数，后面可以跟起始位置和结束位置

# expandtabs

将字符串中tab转换为指定数量的空格

name='Ad\tfjasf'
print  name.expandtabs(5)          #将tab键转换为空格，默认是转换为8个空格

# title

name='this is example'
name.title()          #将首字母变为大写，结果为‘This Is Example’

# strip

移除空白：strip     #只能去除首尾的空白，字符中间的无法去除;另外好像也可以去除换行符

name.strip()
name.lstrip()               #去除左边的空格
name.rstrip()               #去除右边的空格

# split、partition

user,pawd=line.strip().split()

# split返回的结果是list，partition返回的结果是tuple
	rsplit               #从右开始
	rpartition          #和partition类似，左右的区别
	

示例：
    name='alexistoms'
    name.split('is')		 #结果['alex', 'toms']
    name.partition('is') 	 #结果('alex', 'is', 'toms')

# splitlines

name='''
aa
bb
cc'''
print name.splitlines()     #输出结果['', 'aa', 'bb', 'cc']，效果同name.split('\n')

# center、ljust、rjust、zfill

name.center(20,'*')         #将字符串居中显示,效果为‘******Adfjasf*******’
name.ljust(20,'*')			#'abcd****************'
name.rjust(20,'*')			#'****************abcd'

name.zfill(20)          	#字符串补齐到长度20（左侧填0），如果字符串本身的长度已经超过20则不做改变

# startwith、endswith

#返回的结果是布尔值

name.endswith('sf',1,4)               #判断字符串是否已给定字符结尾,可以限定起始和结束位置(1<= <4)
name.startwith     						#以xx开头

# find、index

查找指定字符所在的索引位置

name.find('s')                    #查找字符‘s’在name中的位置，也可以限定开始位置和结束位置
    rfind     #从右向左查找
    rindex     #从右向左查找

name.index('as',2,10)               #和find类似，有点不同：如果字符没有找到，find报-1，index报valueerror

# replice

name.replace('s','S',1)          #替换，name中有两处s，此时指定只替换第一处s

# join

li=['a','l','e','x']
print ''.join(li)               #字符串(此处的'')的拼接，join方法很常用
print '_'.join(li)               #以下划线相隔链接

# maketrans、translate

#一次性替换多个字符串，如a->1, e->2 ...

name.maketrans()      #和translate配合使用，作一个对应表，通过对应表进行替换，2.7没有
name.translate()

示例：
	name = 'this is string example ... wow!!!'
	
    intab='aeiou'			#替换前字符
    outtab='12345'			#替换后字符（a->1, e->2 ...）
    trantab= name.maketrans(intab,outtab)
    name.translate(trantab)

# 字符串格式化输出

# 最常用的方法

  • 方式一

    name= 'alex'
print "i am is %s" % name

  • 方式二

    name='Ad\tfjasf {0} as {1}'
print name.format('xxxx','---')       #format字符串格式化

name="Adfjasf {name} as {id}"
print name.format(name='alex',id='100')     #这样也可以，传递参数
