set集合

一个无序且不重复的元素集合（爬虫的时候会用到）

# 创建set集合

a = {1,2,3}

set()
set(['a','b','c'])			#添加的元素必须时一个可迭代的对象，set(1,2,3)是不可以的
set((1,2,3))

# 去重

# 元素不能重复（能直接对字符串或list去重）

t1=[1,2,1] && set(t1) 								#结果 {1,2}

a='aabbcc' && set(a)								#结果 {'a', 'b', 'c'}

# add

#添加            

a=set() &&  a.add('xx')

# pop、remove

pop：移除元素（好像是随机pop一个元素 ），同时获取被pop的元素
	#a.pop()
	
remove：移除【指定】元素，和pop相比只是移除，没有获取          
	#a.remove('a')

# clear

#清空          

a.clear()

# copy

浅拷贝

# difference

#取出不同，不会更改原始set表

a = set(['a','b','c']) && a.difference(['a']) 		#结果为set(['c', 'b'])，初始的a不变

  • 和symmetric_difference的区别

    #difference 只是将自身元素中的不同作为结果返回，symmetric_difference是将两个集合不同内容合并后作为结果返回

a=set(['a','b','c'])
a.difference(['b','d'])						#结果为{'a', 'c'}
a.symmetric_difference(['b','d'])			#结果为{'a', 'c', 'd'}

# difference_update

#取出不同然后直接修改原set表

a = set(['a','b','c'])
a.difference_update('a')			#此时a的值为{'b', 'c'}

# discard

#移除元素     

a.discard('a')

# intersection

#取交集

#a=set(['a','b','c']) && a.intersection('b')		#结果为{'b'}，此处a.intersection(['b'])也可以

# intersection_update

取交集并修改原set  

a=set(['a','b','c']) && a.intersection_update('b')			#a的值变了

# symmetric_difference

#获取差集，不改变原列表          

b=a.symmetric_difference(['a','b'])

  • 和difference的区别

    #difference 只是将自身元素中的不同作为结果返回，symmetric_difference是将两个集合不同内容合并后作为结果返回

a=set(['a','b','c'])
a.difference(['b','d'])						#结果为{'a', 'c'}
a.symmetric_difference(['b','d'])			#结果为{'a', 'c', 'd'}

# symmetric_difference_update

#获取差集并更新原列表     

a.symmetric_difference_update(['a','b'])

# union

#并集，不修改原列表               

b=a.union(['c','e']), print b

# update

#好像是取并集然后更新原列表          

a.update(['c','e','f'])

# 布尔判断

#返回布尔值

isdisjoint：如果没有交集范围true     				  print a.isdisjoint('b')
issubset ：是否是子集，是则为true                	 	  print a.issubset(['a','b','c','d'])
issuperset ：是否是 超集                              print a.issuperset(['a','b'])
