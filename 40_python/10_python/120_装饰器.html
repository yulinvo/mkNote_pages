      def Before(request,kargs):
    print 'Before'
def After(request,kargs):
    print 'After'

def Filter(before_func,after_func):        #接收的func，此例子中的Before、After
    def outer(main_func):                  #此处main_func才是接收的用户业务函数，此例中的index
        def wrapper(request,kargs):
            before_result=before_func(request,kargs)
            if (before_result != None):
                return  before_result
                
            main_result=main_func(request,kargs)
            if (main_result != None):
                return  main_result
            
            after_result=after_func(request,kargs)
            if(after_result != None):
                return after_result        #疑问同上面return，解释见上面
        return  wrapper
    return outer

@Filter(Before,After)
def index(request,kargs):
    print  'Index'

# 装饰器

装饰器“@xxx”返回的一定是一个函数，而不是执行函数后的返回值！！！！！

def demo2(func):
    print('------demo2------')						#这句加载到内存中时就直接运行了，不需要下面的print(demo1)
    func(1,2,3,a=1,b=2)
    return 12120

@demo2
def demo1(*args,**kwargs):							#demo1 = demo2(demo1)  !!!!!!!
    print('------demo1------',args,kwargs)			#这句加载到内存中时就直接运行了，不需要下面的print(demo1)

# print(demo1)				#此时demo1的值为 12120
